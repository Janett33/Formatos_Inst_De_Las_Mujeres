    <!-- SCRIPT PARA MOSTRAR/OCULTAR MENÚ -->
    <script>
        const bannerMenu = document.getElementById("bannerMenu");
        const submenu = document.getElementById("submenu");

        let menuVisible = false;

        bannerMenu.addEventListener("click", (event) => {
            event.stopPropagation(); // evita cierre inmediato
            menuVisible = !menuVisible;
            submenu.style.display = menuVisible ? "grid" : "none";
        });

        // Evita que clic en submenú lo cierre
        submenu.addEventListener("click", (event) => {
            event.stopPropagation();
        });

        // Cierre al hacer clic fuera
        document.addEventListener("click", () => {
            submenu.style.display = "none";
            menuVisible = false;
        });

	// Cerrar con tecla ESC
	document.addEventListener("keydown", (event) => {
   	if (event.key === "Escape") {
        submenu.style.display = "none";
        menuVisible = false;
    }
});

    </script>
